  function calculate() {

    // Base anchor (your suggested anchor)

    // 2-day Bangalore onsite + custom deck + standard customization

    let content = 175000;

    let delivery = 150000;

    let addons = 0;

    let rights = 0;

    let timelineProcess = 0;

    let locationAdj = 0;

    let score = 0;


    const durationDays = Math.max(1, num("durationDays", 2));

    const locationType = select("locationType");

    const audienceCount = Math.max(1, num("audienceCount", 20));

    const batchType = select("batchType");

    const numBatches = Math.max(1, num("numBatches", 1));


    // Duration scaling (base assumes 2 days)

    if (durationDays !== 2) {

      // content scales less than linearly, delivery scales linearly

      const extraDays = durationDays - 2;

      if (extraDays > 0) {

        content += extraDays * 45000;

        delivery += extraDays * 75000;

        score += extraDays * 4;

      } else {

        content += extraDays * 35000; // reduces if 1 day

        delivery += extraDays * 75000;

        score += extraDays * 3;

      }

    }


    // Multi-batch handling

    if (batchType === "multiple" && numBatches > 1) {

      const extraBatches = numBatches - 1;

      // reuse same content, charge mostly delivery + small adaptation

      delivery += extraBatches * (Math.max(50000, (150000 / 2)));

      content += extraBatches * 10000;

      addons += extraBatches * 5000;

      score += extraBatches * 2;

    }


    // Audience & outcome factors

    const audienceLevel = select("audienceLevel");

    if (audienceLevel === "manager") { content += 20000; score += 3; }

    if (audienceLevel === "mixed") { content += 30000; addons += 10000; score += 5; }


    const industryType = select("industryType");

    if (industryType === "mixed") { content += 25000; score += 4; }


    const programOutcome = select("programOutcome");

    if (programOutcome === "awareness") { content -= 30000; score -= 3; }

    if (programOutcome === "cert_prep") { content += 40000; score += 6; }

    if (programOutcome === "advanced") { content += 60000; score += 8; }


    // Content development scope

    const deckBuild = select("deckBuild");

    if (deckBuild === "adapt") { content -= 70000; score -= 6; }

    else { score += 6; }


    const slideStyle = select("slideStyle");

    if (slideStyle === "premium") { content += 50000; score += 5; }


    const deliverableDepth = select("deliverableDepth");

    if (deliverableDepth === "slides_notes") { content += 35000; score += 4; }


    // Checkboxes: content/features

    if (checked("clientCustomization")) { content += 40000; score += 6; }

    if (checked("aiProcurementUseCases")) { content += 15000; score += 2; }

    if (checked("industryExamples")) { content += 15000; score += 2; }


    if (checked("worksheets")) { addons += 30000; score += 4; }

    if (checked("caseStudies")) { addons += 35000; score += 5; }

    if (checked("simulations")) { addons += 45000; score += 6; }

    if (checked("capstone")) { addons += 30000; score += 4; }

    if (checked("assessment")) { addons += 25000; score += 3; }


    // Delivery expectations

    const deliveryMode = select("deliveryMode");

    if (deliveryMode === "lecture") { delivery -= 20000; score -= 2; }

    if (deliveryMode === "workshop") { delivery += 30000; score += 4; }


    if (checked("activityEveryModule")) { addons += 20000; score += 3; }

    if (checked("participantEvaluation")) { addons += 30000; score += 4; }

    if (checked("stakeholderCall")) { addons += 15000; score += 2; }

    if (checked("dryRun")) { addons += 20000; score += 3; }

    if (checked("postReport")) { addons += 30000; score += 4; }


    // Materials / rights

    if (checked("workbook")) { addons += 35000; score += 4; }

    if (checked("editablePPT")) { rights += 40000; score += 5; }

    if (checked("internalReuse")) { rights += 90000; score += 8; }

    if (checked("otherTrainersUse")) { rights += 75000; score += 7; }

    if (checked("recording")) { rights += 25000; score += 3; }

    if (checked("recordingReuse")) { rights += 90000; score += 8; }


    // Reviews / feedback / timeline

    const reviewRounds = Number(select("reviewRounds"));

    if (reviewRounds > 2) {

      const extraRounds = reviewRounds === 4 ? 2 : (reviewRounds - 2); // "4+" modeled as 2 extra

      timelineProcess += extraRounds * 20000;

      score += extraRounds * 3;

    }


    const feedbackMode = select("feedbackMode");

    if (feedbackMode === "multiple") {

      timelineProcess += 20000;

      score += 3;

    }


    const timeline = select("timeline");

    let rushMultiplier = 1;

    if (timeline === "tight") { rushMultiplier = 1.15; score += 4; }

    if (timeline === "rush") { rushMultiplier = 1.30; score += 7; }


    // Location adjustments

    if (locationType === "india_onsite") {

      locationAdj += 25000;

      score += 2;

    } else if (locationType === "virtual") {

      delivery -= 35000;

      locationAdj -= 15000;

      score -= 1;

    }


    // Participant count scaling (light touch)

    if (audienceCount > 25 && audienceCount <= 40) {

      delivery += 15000;

      score += 2;

    } else if (audienceCount > 40) {

      delivery += 30000;

      addons += 15000;

      score += 4;

    }


    // Floors (avoid unrealistic negatives)

    content = Math.max(60000, content);

    delivery = Math.max(40000, delivery);

    addons = Math.max(0, addons);

    rights = Math.max(0, rights);

    timelineProcess = Math.max(0, timelineProcess);


    // Subtotal before rush multiplier

    let subtotal = content + delivery + addons + rights + timelineProcess + locationAdj;

    let total = subtotal * rushMultiplier;


    // GST

    const gstIncluded = select("gstIncluded") === "yes";

    const totalWithGST = gstIncluded ? total * 1.18 : total;


    // Quote banding

    let band = "Standard";

    let bandClass = "band-standard";

    if (total < 240000) { band = "Lite"; bandClass = "band-lite"; }

    if (total >= 400000) { band = "Premium"; bandClass = "band-premium"; }


    // Display range (±12% typical estimation spread)

    const low = totalWithGST * 0.88;

    const high = totalWithGST * 1.12;


    // Budget comparison

    const clientBudget = num("clientBudget", 0);

    let budgetCompare = "Budget check: n/a";

    if (clientBudget > 0) {

      if (clientBudget >= total) budgetCompare = "Budget check: Within / possible";

      else if (clientBudget >= total * 0.8) budgetCompare = "Budget check: Tight (scope options needed)";

      else budgetCompare = "Budget check: Low vs scope";

    }


    // UI updates

    $("quoteTotal").textContent = INR(totalWithGST);

    $("quoteRange").textContent = `Range: ${INR(low)} – ${INR(high)}`;

    $("bdContent").textContent = INR(content);

    $("bdDelivery").textContent = INR(delivery);

    $("bdAddons").textContent = INR(addons);

    $("bdRights").textContent = INR(rights);

    $("bdTimeline").textContent = INR((subtotal * (rushMultiplier - 1)) + timelineProcess);

    $("bdLocation").textContent = INR(locationAdj);

    $("bdTotal").textContent = INR(totalWithGST);

    $("quoteBandTag").textContent = `Band: ${band}`;

    $("scoreBadge").textContent = `Score: ${Math.max(0, Math.round(score))}`;

    $("gstBadge").textContent = gstIncluded ? "Incl. GST (18%)" : "Excl. GST";

    $("budgetCompare").textContent = budgetCompare;


    $("bandPillWrap").innerHTML = `<span class="pill ${bandClass}">${band} Quote Band</span>`;


    const scoreText =

      band === "Lite"

        ? "Lower-complexity scope: likely adapted content / fewer add-ons / limited rights."

        : band === "Standard"

          ? "Balanced scope: custom deck + delivery with moderate customization and standard rights."

          : "High-complexity scope: customization, add-ons, licensing/reuse, and/or timeline pressure.";


    $("scoreSummary").textContent = scoreText;


    // Summary text for copy

    const summary = buildSummary({

      total: totalWithGST,

      low,

      high,

      band,

      gstIncluded,

      content, delivery, addons, rights,

      timelineCharge: (subtotal * (rushMultiplier - 1)) + timelineProcess,

      locationAdj,

      score: Math.max(0, Math.round(score)),

      clientBudget,

      budgetCompare

    });


    $("summaryOutput").value = summary;

  }


  function buildSummary(data) {

    const lines = [];

    const durationDays = num("durationDays", 2);

    const hoursTotal = num("hoursTotal", 16);

    const locationType = select("locationType");

    const batchType = select("batchType");

    const numBatches = num("numBatches", 1);


    const locationLabel = {

      blr_onsite: "Bangalore onsite",

      india_onsite: "Outside Bangalore (India onsite)",

      virtual: "Virtual"

    }[locationType] || locationType;


    lines.push(`CIPP Training Ballpark Quote (${data.band} band)`);

    lines.push(`----------------------------------------`);

    lines.push(`Estimated total: ${INR(data.total)} ${data.gstIncluded ? "(incl. GST)" : "(+ GST)"}`);

    lines.push(`Range: ${INR(data.low)} – ${INR(data.high)}`);

    lines.push(`Complexity score: ${data.score}`);

    lines.push(``);

    lines.push(`Scope snapshot:`);

    lines.push(`- Duration: ${durationDays} day(s) / ${hoursTotal} hours`);

    lines.push(`- Delivery: ${locationLabel}`);

    lines.push(`- Batching: ${batchType === "multiple" ? `${numBatches} batches` : "Single batch"}`);

    lines.push(`- Audience level: ${$("audienceLevel").options[$("audienceLevel").selectedIndex].text}`);

    lines.push(`- Outcome focus: ${$("programOutcome").options[$("programOutcome").selectedIndex].text}`);

    lines.push(`- Deck: ${$("deckBuild").options[$("deckBuild").selectedIndex].text}`);

    lines.push(`- Slide style: ${$("slideStyle").options[$("slideStyle").selectedIndex].text}`);

    lines.push(`- Deliverable depth: ${$("deliverableDepth").options[$("deliverableDepth").selectedIndex].text}`);

    lines.push(``);

    lines.push(`Breakup (approx):`);

    lines.push(`- Content development: ${INR(data.content)}`);

    lines.push(`- Delivery fee: ${INR(data.delivery)}`);

    lines.push(`- Add-ons / complexity: ${INR(data.addons)}`);

    lines.push(`- Rights / licensing: ${INR(data.rights)}`);

    lines.push(`- Timeline / process surcharge: ${INR(data.timelineCharge)}`);

    lines.push(`- Location adjustment: ${INR(data.locationAdj)}`);

    lines.push(``);

    if (data.clientBudget > 0) {

      lines.push(`Client budget input: ${INR(data.clientBudget)}`);

      lines.push(`${data.budgetCompare}`);

      lines.push(``);

    }

    lines.push(`Recommended next step: Confirm deliverables, rights (editable/reuse/recording), review rounds, and timeline before final quote.`);

    return lines.join("\n");

  }


  calculate();

})();
